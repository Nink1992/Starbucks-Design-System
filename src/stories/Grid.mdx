export const GridBaseInteractive = () => {
  const [hoverPos, setHoverPos] = useState(null);
  const containerRef = useRef(null);

  // Set default center position after component mounts and ref is available
  React.useEffect(() => {
    if (containerRef.current) {
      const rect = containerRef.current.getBoundingClientRect();
      const centerX = rect.width / 2;
      const centerY = rect.height / 2;
      
      const col = Math.floor(centerX / 20);
      const row = Math.floor(centerY / 20);
      
      setHoverPos({ x: col * 20, y: row * 20 });
    }
  }, []);

  const handleMouseMove = (e) => {
    if (!containerRef.current) return;
    const rect = containerRef.current.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    // snap to 20px grid
    const col = Math.floor(x / 20);
    const row = Math.floor(y / 20);
    
    setHoverPos({ x: col * 20, y: row * 20 });
  };

  const handleMouseLeave = () => {
    if (containerRef.current) {
      const rect = containerRef.current.getBoundingClientRect();
      const centerX = rect.width / 2;
      const centerY = rect.height / 2;
      
      const col = Math.floor(centerX / 20);
      const row = Math.floor(centerY / 20);
      
      setHoverPos({ x: col * 20, y: row * 20 });
    }
  };

  return (
    <div 
      className="grid-base-pattern" 
      ref={containerRef}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
    >
      {hoverPos && (
        <div 
          className="grid-base-hover-cell"
          style={{ left: hoverPos.x, top: hoverPos.y }}
        >
          <div className="grid-base-label-top">
            <div className="grid-base-label-top-mark"></div>
            8px
          </div>
          <div className="grid-base-label-right">
            <div className="grid-base-label-right-top-mark"></div>
            8px
          </div>
        </div>
      )}
    </div>
  );
};

# 栅格 Grid

## 网格基数

<div className="section-desc">
  网格基数是栅格系统中的基本网格单位。栅格化之前先定义网格基数尤其重要，一方面规范设计，指导版式设计与内容布局，辅助规范页面元素对齐和间距设定；另一方面节省设计开发沟通的时间。目前栅格系统中以 8 点为网格基数，粒度大小合适，且能够匹配多数主流屏幕。
